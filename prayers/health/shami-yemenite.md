## נוסח שאמי

לְשֵׁם יִחוּד קֻדְשָׁא בְּרִיךְ הוּא וּשְׁכִינְתֵּיהּ, בִּדְחִילוּ וּרְחִימוּ, לְיַחֵד שֵׁם י״ה בְּו״ה בְּיִחוּדָא שְׁלִים, בְּשֵׁם כָּל יִשְׂרָאֵל.

שִׁיר לַמַּעֲלוֹת.  
אֶשָּׂא עֵינַי אֶל הֶהָרִים מֵאַיִן יָבֹא עֶזְרִי.  
עֶזְרִי מֵעִם יי עֹשֵׂה שָׁמַיִם וָאָרֶץ.  
אַל יִתֵּן לַמּוֹט רַגְלֶךָ אַל יָנוּם שֹׁמְרֶךָ.  
הִנֵּה לֹא יָנוּם וְלֹא יִישָׁן שׁוֹמֵר יִשְׂרָאֵל.  
יי שֹׁמְרֶךָ יי צִלְּךָ עַל יַד יְמִינֶךָ.  
יוֹמָם הַשֶּׁמֶשׁ לֹא יַכֶּכָּה וְיָרֵחַ בַּלָּיְלָה.  
יי יִשְׁמָרְךָ מִכָּל רָע יִשְׁמֹר אֶת נַפְשֶׁךָ.  
יי יִשְׁמָר צֵאתְךָ וּבוֹאֶךָ מֵעַתָּה וְעַד עוֹלָם.

מִי שֶׁבֵּרַךְ אֲבוֹתֵינוּ אַבְרָהָם יִצְחָק וְיַעֲקֹב, מֹשֶׁה וְאַהֲרֹן דָּוִד וּשְׁלֹמֹה, הוּא יְבָרֵךְ וְיִשְׁמֹר וְיַצִּיל אֶת כָּל חוֹלֵי עַמּוֹ יִשְׂרָאֵל, וְאֶת כָּל הַצְּרִיכִים רַחֲמִים וּרְפוּאָה. הַקָּדוֹשׁ בָּרוּךְ הוּא יִמָּלֵא רַחֲמִים עֲלֵיהֶם, לְהַחֲזִיקָם וּלְרַפְּאָם, לְחַיּוֹתָם וּלְבָרְכָם, וְיִשְׁלַח לָהֶם מְהֵרָה רְפוּאָה שְׁלֵמָה מִן הַשָּׁמַיִם, רְפוּאַת הַנֶּפֶשׁ וּרְפוּאַת הַגּוּף, בְּתוֹךְ שְׁאָר חוֹלֵי יִשְׂרָאֵל.

אֵל נָא רְפָא נָא לָהּ.

רְפָאֵנוּ יי וְנֵרָפֵא, הוֹשִׁיעֵנוּ וְנִוָּשֵׁעָה, כִּי תְהִלָּתֵנוּ אָתָּה. וְהַעֲלֵה רְפוּאָה שְׁלֵמָה לְכָל מַכּוֹתֵינוּ, כִּי אֵל מֶלֶךְ רוֹפֵא נֶאֱמָן וְרַחֲמָן אָתָּה.

יְהִי רָצוֹן מִלְּפָנֶיךָ יי אֱלֹהֵינוּ וֵאלֹהֵי אֲבוֹתֵינוּ, שֶׁתַּעֲשֶׂה עִמָּנוּ רַחֲמִים וַחֲסָדִים, וְתַרְחִיק מֵעָלֵינוּ כָּל חֳלִי וּמַכְאוֹב וְכָל נֶגַע וּפֶגַע רָע. תֵּן בָּנוּ כֹּחַ וְיִשּׁוּב הַדַּעַת לִשְׁמֹר אֶת הַגּוּף שֶׁנָּתַתָּ בָּנוּ, וּלְהִתְנַהֵג בְּדֶרֶךְ טוֹבָה וִישָׁרָה.

וְזַכֵּנוּ לְחַיִּים טוֹבִים וַאֲרוּכִים, לְשָׁנִים נְעִימוֹת וּמְתֻקָּנוֹת, לְבְרִיאוּת וּלְשָׁלוֹם, לְפַרְנָסָה בְּנַחַת וּלְשִׂמְחַת הַלֵּב; וְאַל יְהֵא טוֹבָתֵנוּ עַל יְדֵי הֶפְסֵד חֲבֵרֵינוּ, וְאַל יְהֵא אֹרֶךְ יָמֵינוּ עַל יְדֵי גְּרָעוֹן חַיֵּי אַחֵרִים, אֶלָּא מִטּוּבְךָ הַגָּדוֹל וּמֵרַחֲמֶיךָ הָרַבִּים.

שְׁמַע קוֹלֵנוּ יי אֱלֹהֵינוּ, חוּס וְרַחֵם עָלֵינוּ, וְקַבֵּל בְּרַחֲמִים וּבְרָצוֹן אֶת תְּפִלָּתֵנוּ. וּמִלְּפָנֶיךָ מַלְכֵּנוּ רֵיקָם אַל תְּשִׁיבֵנוּ, כִּי אַתָּה שׁוֹמֵעַ תְּפִלַּת כָּל פֶּה.

שִׂים שָׁלוֹם טוֹבָה וּבְרָכָה, חֵן וָחֶסֶד וְרַחֲמִים, עָלֵינוּ וְעַל כָּל יִשְׂרָאֵל עַמֶּךָ. בָּרְכֵנוּ אָבִינוּ כֻּלָּנוּ כְּאֶחָד בְּאוֹר פָּנֶיךָ, כִּי בְאוֹר פָּנֶיךָ נָתַתָּ לָּנוּ יי אֱלֹהֵינוּ תּוֹרָה וְחַיִּים, אַהֲבָה וָחֶסֶד, צְדָקָה וְרַחֲמִים, בְּרָכָה וְשָׁלוֹם.

לְךָ לְבַדְּךָ עֵינֵינוּ תְלוּיוֹת, יי אֱלֹהֵינוּ, וּמִמְּךָ לְבַדְּךָ רְפוּאָה וּרְחָמִים. עֲשֵׂה לְמַעַן שְׁמֶךָ, עֲשֵׂה לְמַעַן יְמִינֶךָ, עֲשֵׂה לְמַעַן תּוֹרָתֶךָ, עֲשֵׂה לְמַעַן קְדֻשָּׁתֶךָ.

יִהְיוּ לְרָצוֹן אִמְרֵי פִי וְהֶגְיוֹן לִבִּי לְפָנֶיךָ, יי צוּרִי וְגֹאֲלִי.

עֹשֶׂה שָׁלוֹם בִּמְרוֹמָיו, הוּא בְרַחֲמָיו יַעֲשֶׂה שָׁלוֹם עָלֵינוּ וְעַל כָּל יִשְׂרָאֵל, וְאִמְרוּ אָמֵן.
